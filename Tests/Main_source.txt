procedure wind {
	w1 = 5;
	read w2;
	w3 = 200;
	
	if (w2 < 50) then {
		if (w3 < 10) then {
			if (w1 < 50) then {
				call speed;
			} else {
				call temperature;
			}
		} else {
			if (w1 > 50) then {
                w2 = w1 - 100;
			} else {
                w3 = w1 - w2 % 50;
			}
            w1 = w1 - 100;
		}
	} else {
        print w2;
        w2 = w1 + w3;
	}
	w3 = w1 + w2;
	w2 = w1 / w3 + w2;
	print w3;
}

procedure speed {
    call wind;
    speed = 10;
    distance = 100;
    travelled = speed * distance;
    if (speed > 100) then {
        while (distance > 10000) {
            distance = distance + speed * 2;
        }
    } else {
        distance = distance - speed * 10;
        call wind;
    }
    print speed;
}

procedure humidity {
    read h1;
    read h2;
    read h3;
    h4 = h1 % h2 * h3;
    print h4;
}

procedure temperature {
    call wind;
