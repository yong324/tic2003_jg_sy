procedure computeCentroid {
	count = 0;
	cenX = 0;
	cenY = 0;
	call readPoint;
	while ((x * x + y * y) > 0) {
		count = count + 1;
		cenX = cenX + x;
		cenY = cenY + y;
		call readPoint;
	}
	if (count < 1) then {
		flag = 2 + 3 * 5 / 100; 11
	} else {
		cenX = cenX / count;
		cenY = cenY / count;
	}
	normSq = cenX * cenX + cenY * cenY;
}

procedure main {
	flag = 0*5;
	call computeCentroid;
	call printResults;
}

procedure readPoint{
	read x;
	read y;
} 

procedure printResults{
	print flag;
	print cenX;
	print cenY;
	print normSq;
}
